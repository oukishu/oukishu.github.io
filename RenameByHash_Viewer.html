RenameByHash.ps1 
# Parameters definition
param(
    [string]$TargetDir,
    [string]$Extension
)

# Interactive path retrieval
if ([string]::IsNullOrWhiteSpace($TargetDir)) {
    $TargetDir = Read-Host "Enter target folder path (Press Enter for current directory)"
}
if ([string]::IsNullOrWhiteSpace($TargetDir)) { $TargetDir = Get-Location }

# Interactive extension retrieval
if ([string]::IsNullOrWhiteSpace($Extension)) {
    $Extension = Read-Host "Enter file extension (e.g., png, jpg)"
}
if ([string]::IsNullOrWhiteSpace($Extension)) { $Extension = "png" }
if (!$Extension.StartsWith(".")) { $Extension = "." + $Extension }

Write-Host "`n--- Task Started ---" -ForegroundColor Cyan
Write-Host "Directory: $TargetDir"
Write-Host "Extension: $Extension"

if (!(Test-Path $TargetDir)) {
    Write-Host "Error: Folder does not exist!" -ForegroundColor Red
    pause; exit
}

# Get list of files
$files = Get-ChildItem -Path $TargetDir -Filter "*$Extension" -File

if ($files.Count -eq 0) {
    Write-Host "No matching files found." -ForegroundColor Yellow
} else {
    # Hash Map to store discovered hashes for duplicate detection
    $hashMap = @{}

    foreach ($f in $files) {
        try {
            # Calculate full SHA256 Hash
            $hash = (Get-FileHash $f.FullName -Algorithm SHA256).Hash.ToUpper()
            
            # --- Deduplication Logic ---
            if ($hashMap.ContainsKey($hash)) {
                # If hash exists, it is a duplicate
                $originalFile = $hashMap[$hash]
                Remove-Item -Path $f.FullName -Force
                Write-Host "Deleted Duplicate: $($f.Name) (Identical to $originalFile)" -ForegroundColor Gray
                continue
            }

            # Record new hash in the map
            $hashMap[$hash] = $f.Name

            # --- Renaming Logic ---
            # Use the last 12 characters of the hash as the new filename
            $newNameBase = $hash.Substring($hash.Length - 12)
            $newName = $newNameBase + $Extension
            
            # Skip if the file is already named correctly
            if ($f.Name -eq $newName) { continue }

            # Conflict handling (low probability collision check)
            if (Test-Path (Join-Path $TargetDir $newName)) {
                $count = 1
                while (Test-Path (Join-Path $TargetDir "$newNameBase-$count$Extension")) {
                    $count++
                }
                $newName = "$newNameBase-$count$Extension"
            }

            Rename-Item -Path $f.FullName -NewName $newName -ErrorAction Stop
            Write-Host "Renamed: $($f.Name) -> $newName" -ForegroundColor Green
            
            # Update the map with the new filename
            $hashMap[$hash] = $newName
        }
        catch {
            Write-Host "Failed to process: $($f.Name). Reason: $($_.Exception.Message)" -ForegroundColor Red
        }
    }
}

Write-Host "`n--- Task Completed ---" -ForegroundColor Cyan
pause

README.md

This script renames files based on their SHA256 hash value. It is ideal for organizing wallpaper collections, asset libraries, or identifying duplicate content.

Key Features

  • Content-Based & Deduplication: Identifies files by actual content. If an identical file (matching hash) is found, the duplicate is automatically deleted to save space.
  • Hash Filenames: Uses the last 12 characters of the SHA256 hash for neat and standardized naming.
  • Collision Safety: Implements an incremental counter suffix (e.g., -1, -2) in the extremely rare event of a name collision.
  • Customizable: Prompted inputs for both target directory and file extension.

How to Use

Option A: Direct Execution

  • Click Download above to save the RenameByHash.ps1 file.
  • Right-click the file and select "Run with PowerShell".
  • Follow the on-screen prompts to set the folder path and file type.

Option B: Command Line

Pass parameters directly via the terminal:

  • Safety Notice: This version includes file deletion for duplicate content. Always test on a temporary folder or ensure you have a backup before processing large libraries.
